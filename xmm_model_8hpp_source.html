238  virtual void train(TrainingSet *trainingSet)
239  {
240  if (!trainingSet) return;
241  cancelTraining();
242 
243  is_training_ = true;
244 
245  // Fetch training set parameters
246  shared_parameters->dimension.set(trainingSet->dimension.get());
247  if (shared_parameters->bimodal.get()) {
248  shared_parameters->dimension_input.set(trainingSet->dimension_input.get());
249  }
250  shared_parameters->column_names.set(trainingSet->column_names.get());
251 
252  // Update models
253  bool contLoop(true);
254  while (contLoop) {
255  contLoop = false;
256  for (auto it = models.begin(); it != models.end(); ++it) {
257  if (trainingSet->labels().count(it->first) == 0)
258  {
259  models.erase(it->first);
260  contLoop = true;
261  break;
262  }
263  }
264  }
265  for (typename std::set<std::string>::iterator it=trainingSet->labels().begin(); it != trainingSet->labels().end(); ++it)
266  {
267  addModelForClass(*it);
268  }
269 
270  // Start class training
271  for (auto it=this->models.begin(); it != this->models.end(); ++it) {
272  it->second.is_training_ = true;
273  it->second.cancel_training_ = false;
274  models_still_training_++;
275  if ((configuration.multithreading == MultithreadingMode::Parallel) ||
276  (configuration.multithreading == MultithreadingMode::Background)) {
277  training_threads_[it->first] = std::thread(&SingleClassModel::train,
278  &it->second,
279  trainingSet->getPhrasesOfClass(it->first));
280  } else {
281  it->second.train(trainingSet->getPhrasesOfClass(it->first));
282  }
283  }
284  if (configuration.multithreading == MultithreadingMode::Parallel) {
285  joinTraining();
286  }
287  if (configuration.multithreading == MultithreadingMode::Sequential) {
288  is_training_ = false;
289  }
290  }

474  virtual void joinTraining()
475  {
476  if (!is_training_) return;
477  bool expectedState(false);
478  if (is_joining_.compare_exchange_weak(expectedState, true)) {
479  while (!training_threads_.empty()) {
480  training_threads_.begin()->second.join();
481  training_threads_.erase(training_threads_.begin());
482  }
483  bool classesUntrained(true);
484  while (classesUntrained) {
485  classesUntrained = false;
486  for (auto it = this->models.begin(); it != this->models.end(); it++) {
487  if (it->second.training_status_.status == TrainingEvent::Status::Cancel ||
488  it->second.training_status_.status == TrainingEvent::Status::Error) {
489  models.erase(it);
490  classesUntrained= true;
491  break;
492  }
493  }
494  }
495  TrainingEvent event(this, "", TrainingEvent::Status::Alldone);
496  training_events.notifyListeners(event);
497  is_joining_ = false;
498  is_training_ = false;
499  } else {
500  while (is_joining_) {}
501  }
502  }

507  void onTrainingEvent(TrainingEvent const& e)
508  {
509  event_mutex_.lock();
510  TrainingEvent event(e);
511  event.model = this;
512  training_events.notifyListeners(event);
513  if (e.status != TrainingEvent::Status::Run) {
514  models_still_training_--;
515  ((SingleClassModel *)e.model)->is_training_ = false;
516  if (configuration.multithreading == MultithreadingMode::Background &&
517  !is_joining_ && // avoid to call "joinTraining" if already called by the main thread
518  !cancel_required_ && // avoid to call "joinTraining" if cancel required by the main thread
519  models_still_training_ == 0) {
520  is_joining_ = false;
521  std::thread(&Model<SingleClassModel, ModelType>::joinTraining, this).detach();
522  }
523  }
524  event_mutex_.unlock();
525  }
